//   tclsh embgen.tcl embgen_demo_sample.txt

// ------------------------------------------------------------
// Example 1: echo generator
// ------------------------------------------------------------

//embgen_embedded_generator echo 11111111-1111-1111-1111-111111111111
// This code comes from the note text:
// printf("Hello from embedded generator 1\\n");
// for (int i = 0; i < 3; ++i) {
//     printf("  i = %d\\n", i);
// }
//embgen_generated_start 11111111-1111-1111-1111-111111111111
This code comes from the note text:
printf("Hello from embedded generator 1\\n");
for (int i = 0; i < 3; ++i) {
    printf("  i = %d\\n", i);
}
//embgen_generated_end 11111111-1111-1111-1111-111111111111

// ------------------------------------------------------------
// Example 3: PlantUML ASCII generator
// ------------------------------------------------------------

//embgen_embedded_generator plantuml_ascii 33333333-3333-3333-3333-333333333333
// @startuml
// Alice -> Bob: Hello
// Bob --> Alice: Hi!
// @enduml
//embgen_generated_start 33333333-3333-3333-3333-333333333333
     ,-----.          ,---.
     |Alice|          |Bob|
     `--+--'          `-+-'
        |    Hello      |  
        |-------------->|  
        |               |  
        |     Hi!       |  
        |<- - - - - - - |  
     ,--+--.          ,-+-.
     |Alice|          |Bob|
     `-----'          `---'

//embgen_generated_end 33333333-3333-3333-3333-333333333333


// ------------------------------------------------------------
// Example 4: JSON-driven macro
// ------------------------------------------------------------

//embgen_embedded_generator json_driven_macro 44444444-4444-4444-4444-444444444444
// @types.json {types {name Person} fields} {
//     emit "#{javatype} #{name}; // #{dbtype}\n"
// }
//embgen_generated_start 44444444-4444-4444-4444-444444444444
int id; // INTEGER
String firstName; // VARCHAR(255)
String lastName; // VARCHAR(255)

//embgen_generated_end 44444444-4444-4444-4444-444444444444


// ------------------------------------------------------------
// Example 5: XML-driven macro
// ------------------------------------------------------------

//embgen_embedded_generator xml_driven_macro 55555555-5555-5555-5555-555555555555
// @types.xml {/types/type[@name='Person']/fields/field} {
//     emit "#{javatype} #{name}; // #{dbtype}\n"
// }
//embgen_generated_start 55555555-5555-5555-5555-555555555555
int id; // INTEGER
String firstName; // VARCHAR(255)
String lastName; // VARCHAR(255)
String email; // VARCHAR(255)
String dateOfBirth; // VARCHAR(20)
String phoneNumber; // VARCHAR(50)
String status; // VARCHAR(20)

//embgen_generated_end 55555555-5555-5555-5555-555555555555
